024    package dk.itst.oiosaml.error;
025    
026    /**
027     * Feasible layers to use when throwing unrecoverable exceptions
028     * @author lsteinth
029     */
030    public class Layer {
031        
032            public static final String VERSION = "$Id: Layer.java 2829 2008-05-13 12:11:31Z jre $";
033        public static final Layer CLIENT = new Layer("CLIENT");
034        public static final Layer BUSINESS = new Layer("BUSINESS");
035        public static final Layer DATAACCESS = new Layer("RESOURCE");
036        public static final Layer UNDEFINED = new Layer("UNDEFINED");
037    
038        private String value;
039    
040        /**
041         * Create a new layer
042         * @param value The value of the layer
043         */
044        private Layer(String value) {
045            this.value = value;
046        }
047    
048        /**
049         * 
050         * @return The value of the layer
051         */
052        public String getValue() {
053            return value;
054        }
055    
056        /**
057         * {@inheritDoc Object#toString()}
058         */
059        public String toString() {
060            return value;
061        }
062    
063        /**
064             * @see Object#equals(Object)
065             */
066        public boolean equals(Layer layer) {
067            return value.equals(layer.value);
068        }
069    
070        /**
071         * {@inheritDoc Object#hashCode()}
072         */
073        public int hashCode() {
074            return getValue().hashCode();
075        }
076    
077        /**
078         * {@inheritDoc Object#equals(Object)}
079         */
080        public boolean equals(Object obj) {
081            if(!(obj instanceof Layer)) {
082                return false;
083            } else {
084                return ((getValue() != null) && (getValue().equals(((Layer)obj).getValue())));
085            }
086        }
087    }
