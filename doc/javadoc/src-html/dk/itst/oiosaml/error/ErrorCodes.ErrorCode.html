024    package dk.itst.oiosaml.error;
025    
026    /**
027     * Feasible error codes to use
028     * @author lsteinth
029     */
030    public class ErrorCodes {
031    
032            public static final String VERSION = "$Id: ErrorCodes.java 2829 2008-05-13 12:11:31Z jre $";
033        public static final ErrorCode RUNTIME_EXCEPTION = new ErrorCode("RUNTIME_EXCEPTION");
034        public static final ErrorCode REMOTE_EXCEPTION = new ErrorCode("REMOTE_EXCEPTION");
035        public static final ErrorCode WRAPPED_EXCEPTION = new ErrorCode("WRAPPED_EXCEPTION");
036        public static final ErrorCode SERIALIZED_EXCEPTION = new ErrorCode("SERIALIZED_EXCEPTION");
037        public static final ErrorCode OBJECT_NOT_FOUND = new ErrorCode("OBJECT_NOT_FOUND");
038        public static final ErrorCode STARTUP_EXCEPTION = new ErrorCode("STARTUP_EXCEPTION");
039    
040        public static class ErrorCode {
041            private String value;
042    
043            /**
044             * Create a new <code>ErrorCode</code>
045             * @param value The value of the error
046             */
047            public ErrorCode(String value) {
048                this.value = value;
049            }
050    
051            /**
052             * @return The value of the error code
053             */
054            public String getValue() {
055                return value;
056            }
057        }
058    
059    }
