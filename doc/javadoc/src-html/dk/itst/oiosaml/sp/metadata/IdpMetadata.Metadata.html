071    public class IdpMetadata {
072            public static final String VERSION = "$Id: IdpMetadata.java 2964 2008-06-02 11:34:06Z jre $";
073            public static final String METADATA_DIRECTORY = "common.saml2.metadata.idp.directory";
074            private static IdpMetadata instance;
075    
076            private static final Logger log = Logger.getLogger(IdpMetadata.class);
077    
078            private final Map<String, Metadata> metadata = new HashMap<String, Metadata>();
079    
080            public IdpMetadata(String protocol, EntityDescriptor ... entityDescriptor) {
081                    for (EntityDescriptor descriptor : entityDescriptor) {
082                            if (metadata.containsKey(descriptor.getEntityID())) {
083                                    metadata.get(descriptor.getEntityID()).addCertificates(new Metadata(descriptor, protocol).getCertificates());
084                            } else {
085                                    metadata.put(descriptor.getEntityID(), new Metadata(descriptor, protocol));
086                            }
087                    }
088            }
089    
090            public static IdpMetadata getInstance() {
091                    if (instance == null) {
092                            Configuration conf = SAMLConfiguration.getSystemConfiguration();
093                            String directory = SAMLConfiguration.getStringPrefixedWithBRSHome(conf, METADATA_DIRECTORY);
094                            File idpDir = new File(directory);
095                            File[] files = idpDir.listFiles(new FilenameFilter() {
096                                    public boolean accept(File dir, String name) {
097                                            return name.toLowerCase().endsWith(".xml");
098                                    }
099                            });
100                            String protocol = conf.getString(Constants.PROP_PROTOCOL);
101                            
102                            List<EntityDescriptor> descriptors = new ArrayList<EntityDescriptor>();
103                            for (File md : files) {
104                                    log.info("Loading " + protocol + " metadata from " + md);
105                                    try {
106                                            XMLObject descriptor = SAMLUtil.unmarshallElementFromFile(md.getAbsolutePath());
107                                            if (descriptor instanceof EntityDescriptor) {
108                                                    descriptors.add((EntityDescriptor) descriptor);
109                                            } else if (descriptor instanceof EntitiesDescriptor) {
110                                                    EntitiesDescriptor desc = (EntitiesDescriptor) descriptor;
111                                                    descriptors.addAll(desc.getEntityDescriptors());
112                                            } else {
113                                                    throw new RuntimeException("Metadata file " + md + " does not contain an EntityDescriptor. Found " + descriptor.getElementQName()  + ", expected " + EntityDescriptor.ELEMENT_QNAME);
114                                            }
115                                    } catch (RuntimeException e) {
116                                            log.error("Unable to load metadata from " + md + ". File must contain valid XML and have EntityDescriptor as top tag",e);
117                                            throw e;
118                                    }
119                            }
120                            if (descriptors.isEmpty()) {
121                                    throw new IllegalStateException("No IdP descriptors found in " + directory + "! At least one file is required.");
122                            }
123                            instance = new IdpMetadata(protocol, descriptors.toArray(new EntityDescriptor[descriptors.size()]));
124                    }
125                    return instance ;
126            }
127            
128            public static void setMetadata(IdpMetadata metadata) {
129                    instance = metadata;
130            }
131    
132            public Metadata getMetadata(String entityID) {
133                    Metadata md = metadata.get(entityID);
134                    if (md == null) {
135                            throw new IllegalArgumentException("No metadata found for " + entityID);
136                    }
137                    return md;
138            }
139    
140            /**
141             * Check if SAML Discovery Profile should be enabled.
142             * 
143             * If more than one metadata file exists, discovery should be enabled, and this method will return true.
144             */
145            public boolean enableDiscovery() {
146                    return metadata.size() > 1;
147            }
148    
149            /**
150             * Get the first registered metadata.
151             * 
152             * This method should only be used when {@link #enableDiscovery()} returns <code>true</code>, as the 
153             * metadata list is not ordered.
154             */
155            public Metadata getFirstMetadata() {
156                    return getMetadata(getEntityIDs().iterator().next());
157            }
158    
159            public Collection<String> getEntityIDs() {
160                    return metadata.keySet();
161            }
162    
163    
164            public Metadata findSupportedEntity(String ... entityIds) {
165                    for (String entityId : entityIds) {
166                            Metadata md = metadata.get(entityId);
167                            if (md != null) {
168                                    return md;
169                            }
170                    }
171                    throw new IllegalArgumentException("No supported idp found in " + Arrays.toString(entityIds) + ". Supported ids: " + metadata.keySet());
172            }

174            public static class Metadata {
175                    private EntityDescriptor entityDescriptor;
176                    private IDPSSODescriptor idpSSODescriptor;
177                    private Collection<X509Certificate> certificates = new ArrayList<X509Certificate>();
178                    private Collection<X509Certificate> validCertificates = new HashSet<X509Certificate>();
179    
180                    private Metadata(EntityDescriptor entityDescriptor, String protocol) {
181                            this.entityDescriptor = entityDescriptor;
182                            idpSSODescriptor = entityDescriptor.getIDPSSODescriptor(protocol);
183                            try {
184                                    X509Certificate cert = SecurityHelper.buildJavaX509Cert(getCertificateNode().getValue());
185                                    certificates.add(cert);
186                                    validCertificates.add(cert);
187                            } catch (CertificateException e) {
188                                    throw new WrappedException(Layer.BUSINESS, e);
189                            }
190                    }
191    
192    
193                    public void addCertificates(Collection<X509Certificate> certificates) {
194                            this.certificates.addAll(certificates);
195                            this.validCertificates.addAll(certificates);
196                    }
197    
198    
199                    /**
200                     * 
201                     * @return The entityID of the Login Site
202                     */
203                    public String getEntityID() {
204                            return entityDescriptor.getEntityID();
205                    }
206    
207                    /**
208                     * 
209                     * @return The location (URL) of {@link ArtifactResolutionService}.
210                     */
211                    public String getArtifactResolutionServiceLocation(String binding) throws IllegalArgumentException {
212                            for (ArtifactResolutionService artifactResolutionService : idpSSODescriptor.getArtifactResolutionServices()) {
213                                    if (SAMLConstants.SAML2_SOAP11_BINDING_URI.equals(artifactResolutionService.getBinding())) {
214                                            return artifactResolutionService.getLocation();
215                                    }
216                            }
217                            throw new IllegalArgumentException("No artifact resolution service for binding " + binding);
218                    }
219    
220                    /**
221                     * Get a signon service location for a specific binding.
222                     * @param binding SAML binding name,
223                     * @return The url for the location.
224                     * @throws IllegalArgumentException if the binding is not present in metadata.
225                     */
226                    public String getSingleSignonServiceLocation(String binding) throws IllegalArgumentException {
227                            for (SingleSignOnService service : idpSSODescriptor.getSingleSignOnServices()) {
228                                    if (service.getBinding().equals(binding)) {
229                                            return service.getLocation();
230                                    }
231                            }
232                            throw new IllegalArgumentException("Binding " + binding + " not found");
233                    }
234                    
235                    public String getAttributeQueryServiceLocation(String binding) throws IllegalArgumentException {
236                            AttributeAuthorityDescriptor descriptor = entityDescriptor.getAttributeAuthorityDescriptor(SAMLConstants.SAML20P_NS);
237                            if (descriptor == null) throw new IllegalArgumentException("Metadata does not contain a AttributeAuthorityDescriptor");
238                            for (AttributeService service : descriptor.getAttributeServices()) {
239                                    if (binding.equals(service.getBinding())) {
240                                            return service.getLocation();
241                                    }
242                            }
243                            throw new IllegalArgumentException("Binding " + binding + " not found in AttributeServices");
244                    }
245    
246                    public List<SingleSignOnService> getSingleSignonServices() {
247                            return idpSSODescriptor.getSingleSignOnServices();
248                    }
249                    /**
250                     * 
251                     * @return The location (URL) of {@link SingleSignOnService} at the Login Site
252                     */
253                    public String getSingleLogoutServiceLocation() {
254                            String url = null;
255                            if (idpSSODescriptor.getSingleLogoutServices().size() > 0) {
256                                    SingleLogoutService singleLogoutService = idpSSODescriptor.getSingleLogoutServices().get(0);
257                                    url = singleLogoutService.getLocation();
258                            }
259                            return url;
260                    }
261    
262                    /**
263                     * 
264                     * @return The response location (URL) of {@link SingleSignOnService} at the Login Site
265                     */
266                    public String getSingleLogoutServiceResponseLocation() {
267                            if (idpSSODescriptor.getSingleLogoutServices().size() > 0) {
268                                List<SingleLogoutService> singleLogoutServices = idpSSODescriptor.getSingleLogoutServices();
269    
270                    // Prefer POST binding - due to browser redirect limitations.
271                                SingleLogoutService singleLogoutService = idpSSODescriptor.getSingleLogoutServices().get(0);
272                                for (SingleLogoutService sls : singleLogoutServices) {
273                                    if(sls.getBinding().equals(SAMLConstants.SAML2_POST_BINDING_URI)) {
274                                        singleLogoutService = sls;
275                                        break;
276                                    }
277                    }
278    
279                                String location = singleLogoutService.getResponseLocation();
280                                    if (location == null) {
281                                            location = singleLogoutService.getLocation();
282                                    }
283                                    return location;
284                            }
285                            return null;
286                    }
287    
288    
289                    /**
290                     * 
291                     * @return The certificate node from the metadata associated with the Login
292                     *         Site
293                     */
294                    private org.opensaml.xml.signature.X509Certificate getCertificateNode() {
295                            if (idpSSODescriptor != null && idpSSODescriptor.getKeyDescriptors().size() > 0) {
296                                    KeyDescriptor keyDescriptor = idpSSODescriptor.getKeyDescriptors().get(0);
297                                    if (keyDescriptor.getKeyInfo().getX509Datas().size() > 0) {
298                                            X509Data x509Data = keyDescriptor.getKeyInfo().getX509Datas().get(0);
299                                            if (x509Data.getX509Certificates().size() > 0) {
300                                                    return x509Data.getX509Certificates().get(0);
301                                            }
302                                    }
303                            }
304                            throw new IllegalStateException("IdP Metadata does not contain a certificate: " + getEntityID());
305                    }
306                    
307                    Collection<X509Certificate> getAllCertificates() {
308                            return certificates;
309                    }
310    
311                    /**
312                     * Get a list of all valid certificates for this IdP.
313                     * 
314                     * Any expired or revoked certificates will not be included in the list. 
315                     */
316                    public Collection<X509Certificate> getCertificates() {
317                            Collection<X509Certificate> res = new ArrayList<X509Certificate>();
318                            for (X509Certificate certificate : validCertificates) {
319                                    if (certificate.getNotAfter().after(new Date())) {
320                                            res.add(certificate);
321                                    } else {
322                                            log.debug("Local Metadata certificate for " + getEntityID() + " expired at " + certificate.getNotAfter() + ", current: " + new Date());
323                                    }
324                            }
325                            return res;
326                    }
327    
328                    void setCertificateValid(X509Certificate cert, boolean valid) {
329                            if (valid) {
330                                    validCertificates.add(cert);
331                            } else {
332                                    validCertificates.remove(cert);
333                            }
334                    }

337                    /**
338                     * Find a supported login endpoint.
339                     * @throws IllegalArgumentException If no services match the selected bindings. 
340                     */
341                    public Endpoint findLoginEndpoint(String[] bindings) {
342                            if (bindings == null) throw new IllegalArgumentException("bindings cannot be null");
343                            
344                            for (String binding : bindings) {
345                                    for (SingleSignOnService service : idpSSODescriptor.getSingleSignOnServices()) {
346                                            if (service.getBinding().equalsIgnoreCase(binding)) {
347                                                    return service;
348                                            }
349                                    }
350                            }
351                            throw new IllegalArgumentException("No SingleSignonService found for " + Arrays.toString(bindings));
352                    }

354                    /**
355                     * Get the name format for an attribute.
356                     * 
357                     * @param attribute The attribute to look for.
358                     * @param defaultFormat The format to return if the attribute is not present in idp metadata.
359                     */
360                    public String getAttributeNameFormat(String attribute, String defaultFormat) {
361                            for (Attribute attr : idpSSODescriptor.getAttributes()) {
362                                    if (attribute.equals(attr.getName())) {
363                                            return attr.getNameFormat();
364                                    }
365                            }
366                            return defaultFormat;
367                    }

369    
370                    public Collection<PublicKey> getPublicKeys() {
371                            Collection<PublicKey> res = new ArrayList<PublicKey>();
372                            for (X509Certificate cert : getCertificates()) {
373                                    res.add(cert.getPublicKey());
374                            }
375                            return res;
376                    }
