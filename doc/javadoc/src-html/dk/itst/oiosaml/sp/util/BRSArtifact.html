024    package dk.itst.oiosaml.sp.util;
025    
026    import java.io.UnsupportedEncodingException;
027    import java.security.MessageDigest;
028    import java.security.NoSuchAlgorithmException;
029    import java.security.SecureRandom;
030    import java.util.Arrays;
031    
032    import org.opensaml.common.binding.BindingException;
033    import org.opensaml.saml2.binding.artifact.AbstractSAML2Artifact;
034    import org.opensaml.saml2.binding.artifact.SAML2ArtifactBuilderFactory;
035    import org.opensaml.saml2.binding.artifact.SAML2ArtifactType0004;
036    import org.opensaml.xml.util.Base64;
037    import org.opensaml.xml.util.DatatypeHelper;
038    
039    import dk.itst.oiosaml.common.OIOSAMLConstants;
040    import dk.itst.oiosaml.error.Layer;
041    import dk.itst.oiosaml.error.WrappedException;
042    
043    /**
044     * Utility methods related to extract the content of a SAML Artifact
045     * @author lsteinth
046     *
047     */
048    public class BRSArtifact {
049            public static final String VERSION = "$Id: BRSArtifact.java 2829 2008-05-13 12:11:31Z jre $";
050            private static SAML2ArtifactBuilderFactory artifactFactory = new SAML2ArtifactBuilderFactory();
051            
052        private SAML2ArtifactType0004 samlArtifact = null;
053        private final String entityId;
054    
055        /**
056             * Build the class from a base64 encoded SAML artifact
057             * 
058             * @param samlArt
059             *            The base64 encoded SAML artifact received from the Login Site
060             * @param relyingParty
061             *            The expected EntityID of the Login Site who issued the SAML artifact
062             * @throws BindingException
063             *             If the SAML artifact is not valid, e.g. does not come from
064             *             the <code>relyingParty</code>
065             */
066        public BRSArtifact(String samlArt, String ... relyingParty) throws BindingException {
067            samlArtifact = decodeArtifact(samlArt);
068            entityId = validate(relyingParty);
069        }
070        
071        /**
072         * Decode the SAML artifact
073         * @param samlArt The base64 encoded SAML artifact
074         * @return The decoded SAML artifact
075         * @throws BindingException If the SAML artifact is not valid
076         */
077        private SAML2ArtifactType0004 decodeArtifact(String samlArt) throws BindingException {
078                    byte[] artifactBytes = Base64.decode(samlArt);
079                    AbstractSAML2Artifact artifact = artifactFactory.buildArtifact(artifactBytes);
080                    if (artifact instanceof SAML2ArtifactType0004) {
081                            return (SAML2ArtifactType0004) artifact;
082                    } else {
083                            throw new BindingException("The artifact is not of the expected type: SAML2ArtifactType004");
084                    }
085        }
086        
087        /**
088             * Check whether the SAML artifact is valid
089             * 
090             * @param relyingParties
091             *            The expected EntityID of the Login Site who issued the SAML
092             *            artifact If the SAML artifact is not valid, e.g. does not come
093             *            from the <code>relyingParty</code>
094             */
095            private String validate(String ... relyingParties) throws BindingException {
096                    byte[] sourceID = samlArtifact.getSourceID();
097                    try {
098                            for (String entityId : relyingParties) {
099                                    MessageDigest md = MessageDigest.getInstance(OIOSAMLConstants.SHA_HASH_ALGORHTM);
100                            byte[] expectedSourceID = md.digest(entityId.getBytes(OIOSAMLConstants.UTF_8));
101    
102                            if (Arrays.equals(expectedSourceID, sourceID)) {
103    
104                                    return entityId;
105                            }
106                            }
107    
108                            throw new BindingException("The sourceID:"+new String(sourceID)+" does not match the expected sourceId");
109                    } catch (NoSuchAlgorithmException e) {
110                            throw new WrappedException(Layer.DATAACCESS, e);
111                    } catch (UnsupportedEncodingException e) {
112                            throw new WrappedException(Layer.DATAACCESS, e);
113                    }
114            }
115    
116            /**
117             * Extract the endpointIndex of the SAML artifact which pin points the URL
118             * of the artifact resolution service
119             * 
120             * @return The endpointIndex
121             */
122        public int getEndpointIndex() {
123                    byte[] endpointIndex = samlArtifact.getEndpointIndex();
124                    //Convert the endpointIndex to an integer
125                    int endpointIndexValue = 0;
126            for (int i = 0; i < endpointIndex.length; i++) {
127                int shift = (endpointIndex.length - 1 - i) * 8;
128                endpointIndexValue += (endpointIndex[i] & 0x000000FF) << shift;
129            }
130            return endpointIndexValue;              
131        }
132    
133        /**
134         * 
135         * @param endPointIndex The endPointIndex to be used
136         * @param entityID The entityID of the Identity Provider
137         * @return An artifact for the given enitiyID
138         */
139        public static SAML2ArtifactType0004 buildArtifact(int endPointIndex, String entityID) {
140            try {
141                byte[] endpointIndex = DatatypeHelper.intToByteArray(endPointIndex);
142                byte[] trimmedIndex = new byte[2];
143                trimmedIndex[0] = endpointIndex[2];
144                trimmedIndex[1] = endpointIndex[3];
145    
146                MessageDigest sha1Digester = MessageDigest.getInstance(OIOSAMLConstants.SHA_HASH_ALGORHTM);
147                byte[] source = sha1Digester.digest(entityID.getBytes());
148    
149                SecureRandom handleGenerator = SecureRandom.getInstance("SHA1PRNG");
150                byte[] assertionHandle;
151                assertionHandle = new byte[20];
152                handleGenerator.nextBytes(assertionHandle);
153    
154                return new BRSArtifactType0004(trimmedIndex, source, assertionHandle);
155            } catch (NoSuchAlgorithmException e) {
156                throw new InternalError("JVM does not support required cryptography algorithms: SHA-1/SHA1PRNG.");
157            }
158        }
159        
160        /**
161         * Get the id of the entity which sent the artifact.
162         */
163        public String getEntityId() {
164                    return entityId;
165            }
166    }
