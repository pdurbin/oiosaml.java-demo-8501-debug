024    package dk.itst.oiosaml.sp.util;
025    
026    import java.util.Collections;
027    import java.util.List;
028    
029    public class LogoutRequestValidationException extends Exception {
030    
031            private static final long serialVersionUID = -1726459502073878835L;
032            private final List<String> errors;
033    
034            public LogoutRequestValidationException(List<String> errors) {
035                    super("Validation errors: " + errors);
036                    this.errors = errors;
037            }
038            
039            public List<String> getErrors() {
040                    return Collections.unmodifiableList(errors);
041            }
042    
043    }
