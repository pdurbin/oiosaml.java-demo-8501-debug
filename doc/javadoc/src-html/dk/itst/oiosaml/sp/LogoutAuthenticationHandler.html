024    package dk.itst.oiosaml.sp;
025    
026    import javax.servlet.http.HttpServletRequest;
027    import javax.servlet.http.HttpServletResponse;
028    
029    /**
030     * Use this interface to receive notifications about user authentications and logouts. 
031     * 
032     * @author Joakim Recht
033     *
034     */
035    public interface LogoutAuthenticationHandler extends AuthenticationHandler {
036    
037            /**
038             * Receive a notification when the user is logged out.
039             * 
040             * Nothing should be written to the response under normal circumstances.
041             * 
042             * The method might be called twice during a SLO, depending on the IdP behavior.
043             */
044            public void userLoggedOut(HttpServletRequest request, HttpServletResponse response);
045            
046    }
