084    /**
085     * Configuration hander for initial OIOSAML-J configuration.
086     * 
087     * <p>This servlet is used when the system has not yet been configured correctly. It checks
088     * if oiosaml-j.home points to an existing dir. If the dir is empty, configuration can proceed. As a
089     * security precaution, configuration can never proceed if the configuration dir is not empty.</p>
090     * 
091     * <p>On a normal GET, a page with a form is displayed. SAML EntityID and endpoint URLs are
092     * autodetected based on the requested url.</p>
093     * 
094     * <p>The user can upload IdP metadata, SP certificate, and fill out some additional required info.
095     * When the completed form is POSTed, a new SP metadata file is generated together with the
096     * necessary property files. All of this is written to a ZIP-file, which can be downloaded. If the home
097     * dir is writable, the generated files will also be placed there automatically.</p>
098     * 
099     * @author Joakim Recht <jre@trifork.com>
100     *
101     */

108            public ConfigurationHandler(ServletContext servletContext) {
109                    this.servletContext = servletContext;

111                    engine.setProperty(VelocityEngine.RESOURCE_LOADER, "classpath");
112                    engine.setProperty("classpath.resource.loader.class", "org.apache.velocity.runtime.resource.loader.ClasspathResourceLoader");
113                    try {
114                            engine.init();
115                    } catch (Exception e) {
116                            log.error("Unable to initialize Velocity", e);
117                            throw new WrappedException(Layer.BUSINESS, e);
118                    }

121            public void handleGet(RequestContext context) throws ServletException, IOException {
122                    HttpServletRequest request = context.getRequest();
123                    HttpServletResponse response = context.getResponse();
124                    if (request.getParameter("download") != null) {
125                            byte[] conf = (byte[]) request.getSession().getAttribute(SESSION_CONFIGURATION);
126                            if (conf == null) {
127                                    response.sendError(HttpServletResponse.SC_NOT_FOUND, "No configuration available for download");
128                                    return;
129                            }
130                            response.setContentType("application/octet-stream");
131                            response.setContentLength(conf.length);
132                            response.addHeader("Content-disposition", "attachment; filename=oiosaml.java-config.zip");
133                            response.getOutputStream().write(conf);
134                            return;
135                    }
136                    if (!checkConfiguration(response)) return;
137                    
138                    Map<String, Object> params = getStandardParameters(request);
139                    
140                    String res = renderTemplate("configure.vm", params, true);
141                    sendResponse(response, res);
142            }

145            public void handlePost(RequestContext context) throws ServletException, IOException {
146                    HttpServletRequest request = context.getRequest();
147                    HttpServletResponse response = context.getResponse();
148                    
149                    if (!checkConfiguration(response)) return;
150                    
151                    List<?> parameters = extractParameterList(request);
152                    
153                    String orgName = extractParameter("organisationName", parameters);
154                    String orgUrl = extractParameter("organisationUrl", parameters);
155                    String email = extractParameter("email", parameters);
156                    String entityId = extractParameter("entityId", parameters);
157                    final String password = extractParameter("keystorePassword", parameters);
158                    byte[] metadata = extractFile("metadata", parameters).get();
159                    FileItem ksData = extractFile("keystore", parameters);
160                    byte[] keystore = null;
161                    if (ksData != null) {
162                            keystore = ksData.get();
163                    }
164                    if (!checkNotNull(orgName, orgUrl, email, password, metadata, entityId)
165                                    || metadata.length == 0
166                                    || (keystore == null && !Boolean.valueOf(extractParameter("createkeystore", parameters)))) {
167                            Map<String, Object> params = getStandardParameters(request);
168                            params.put("error", "All fields must be filled.");
169                            params.put("organisationName", orgName);
170                            params.put("organisationUrl", orgUrl);
171                            params.put("email", email);
172                            params.put("keystorePassword", password);
173                            params.put("entityId", entityId);
174                            log.info("Parameters not correct: " + params);
175                            log.info("Metadata: " + new String(metadata));
176                            
177                            String res = renderTemplate("configure.vm", params, true);
178                            sendResponse(response, res);
179                            return;
180                    }
181                    
182                    Credential credential = context.getCredential();
183                    if (keystore != null && keystore.length > 0) {
184                            credential  = CredentialRepository.createCredential(new ByteArrayInputStream(keystore), password);
185                    } else if (Boolean.valueOf(extractParameter("createkeystore", parameters))) {
186                            try {
187                                    BasicX509Credential cred = new BasicX509Credential();
188                                    KeyPair kp = dk.itst.oiosaml.security.SecurityHelper.generateKeyPairFromURI("http://www.w3.org/2001/04/xmlenc#rsa-1_5", 1024);
189                                    cred.setPrivateKey(kp.getPrivate());
190                                    cred.setPublicKey(kp.getPublic());
191                                    credential = cred;
192                                    
193                                    KeyStore ks = KeyStore.getInstance("JKS");
194                                    ks.load(null, null);
195                                    X509Certificate cert = dk.itst.oiosaml.security.SecurityHelper.generateCertificate(credential, getEntityId(request));
196                                    cred.setEntityCertificate(cert);
197                                    
198                                    ks.setKeyEntry("oiosaml", credential.getPrivateKey(), password.toCharArray(), new Certificate[] { cert });
199                                    ByteArrayOutputStream bos = new ByteArrayOutputStream();
200                                    ks.store(bos, password.toCharArray());
201                                    
202                                    keystore = bos.toByteArray();
203                                    bos.close();
204                            } catch (Exception e) {
205                    log.error("Unable to generate credential", e);
206                                    throw new RuntimeException("Unable to generate credential", e);
207                            }
208                    }
209                    
210                    EntityDescriptor descriptor = generateSPDescriptor(getBaseUrl(request), entityId, credential, orgName, orgUrl, email, 
211                                    Boolean.valueOf(extractParameter("enableArtifact", parameters)), Boolean.valueOf(extractParameter("enablePost", parameters)), 
212                                    Boolean.valueOf(extractParameter("enableSoap", parameters)),
213                    Boolean.valueOf(extractParameter("enablePostSLO", parameters)),
214                                    Boolean.valueOf(extractParameter("supportOCESAttributeProfile", parameters)));
215                    File zipFile = generateZipFile(request.getContextPath(), password, metadata, keystore, descriptor);
216                    
217                    byte[] configurationContents = saveConfigurationInSession(request, zipFile);
218                    boolean written = writeConfiguration(getHome(servletContext), configurationContents);
219                    
220                    Map<String, Object> params = new HashMap<String, Object>();
221                    params.put("home", getHome(servletContext));
222                    params.put("written", written);
223                    sendResponse(response, renderTemplate("done.vm", params, true));
224            }

226            public boolean writeConfiguration(String homeDir, byte[] configurationContents) {
227                    File root = new File(homeDir);
228                    if (!root.isDirectory() || !root.canWrite()) {
229                            return false;
230                    }
231                    boolean written = true;
232                    try {
233                            ZipInputStream input = new ZipInputStream(new ByteArrayInputStream(configurationContents));
234                            ZipEntry next = null;
235                            while ((next = input.getNextEntry()) != null) {
236                                    File newFile = new File(root, next.getName());
237                                    FileUtils.forceMkdir(newFile.getParentFile());
238                                    
239                                    FileOutputStream file = new FileOutputStream(newFile);
240                                    IOUtils.copy(input, file);
241                                    file.close();
242                                    input.closeEntry();
243                            }
244                            input.close();
245                    } catch (IOException e) {
246                            log.error("Unable to write configuration files to " + root , e);
247                            written = false;
248                    }
249                    return written;
250            }

252            private byte[] saveConfigurationInSession(final HttpServletRequest request,
253                            File zipFile) throws IOException, FileNotFoundException {
254                    byte[] configurationContents = IOUtils.toByteArray(new FileInputStream(zipFile));
255                    request.getSession().setAttribute(SESSION_CONFIGURATION, configurationContents);
256                    return configurationContents;
257            }

259            protected File generateZipFile(final String contextPath, final String password, byte[] idpMetadata, byte[] keystore, EntityDescriptor descriptor) throws IOException {
260                    File zipFile = File.createTempFile("oiosaml-", ".zip");
261                    ZipOutputStream zos = new ZipOutputStream(new FileOutputStream(zipFile));
262                    zos.putNextEntry(new ZipEntry("oiosaml-sp.properties"));
263                    zos.write(renderTemplate("defaultproperties.vm", new HashMap<String, Object>() {{
264                            put("homename", Constants.PROP_HOME);
265    
266                            put("servletPath", contextPath);
267                            put("password", password);
268                    }}, false).getBytes());
269                    zos.closeEntry();
270                    
271                    zos.putNextEntry(new ZipEntry("metadata/SP/SPMetadata.xml"));
272                    zos.write(SAMLUtil.getSAMLObjectAsPrettyPrintXML(descriptor).getBytes());
273                    zos.closeEntry();
274                    
275                    zos.putNextEntry(new ZipEntry("metadata/IdP/IdPMetadata.xml"));
276                    zos.write(idpMetadata);
277                    zos.closeEntry();
278                    
279                    zos.putNextEntry(new ZipEntry("certificate/keystore"));
280                    zos.write(keystore);
281                    zos.closeEntry();
282                    
283                    zos.putNextEntry(new ZipEntry("oiosaml-sp.log4j.xml"));
284                    IOUtils.copy(getClass().getResourceAsStream("oiosaml-sp.log4j.xml"), zos);
285                    zos.closeEntry();
286                    
287                    zos.close();
288                    return zipFile;
289            }

291            protected EntityDescriptor generateSPDescriptor(String baseUrl, String entityId, Credential credential, String orgName, String orgUrl, String email, boolean enableArtifact, boolean enableRedirect, boolean enableSoap, boolean enablePostSLO, boolean supportOCESAttributes) {
292                    EntityDescriptor descriptor = SAMLUtil.buildXMLObject(EntityDescriptor.class);
293                    descriptor.setEntityID(entityId);
294                    
295                    SPSSODescriptor spDescriptor = SAMLUtil.buildXMLObject(SPSSODescriptor.class);
296                    spDescriptor.setAuthnRequestsSigned(true);
297                    spDescriptor.setWantAssertionsSigned(true);
298                    
299                    ContactPerson contact = SAMLUtil.buildXMLObject(ContactPerson.class);
300                    contact.getEmailAddresses().add(SAMLUtil.createEmail(email));
301                    contact.setCompany(SAMLUtil.createCompany(orgName));
302                    contact.setType(ContactPersonTypeEnumeration.TECHNICAL);
303                    
304                    descriptor.getContactPersons().add(contact);
305                    descriptor.setOrganization(SAMLUtil.createOrganization(orgName, orgName, orgUrl));
306                    
307                    KeyDescriptor signingDescriptor = SAMLUtil.buildXMLObject(KeyDescriptor.class);
308                    signingDescriptor.setUse(UsageType.SIGNING);
309                    KeyDescriptor encryptionDescriptor = SAMLUtil.buildXMLObject(KeyDescriptor.class);
310                    encryptionDescriptor.setUse(UsageType.ENCRYPTION);
311    
312                    try {
313                            KeyInfoGenerator gen = SecurityHelper.getKeyInfoGenerator(credential, org.opensaml.xml.Configuration.getGlobalSecurityConfiguration(), null);
314                            signingDescriptor.setKeyInfo(gen.generate(credential));
315                            encryptionDescriptor.setKeyInfo(gen.generate(credential));
316                    } catch (SecurityException e1) {
317                            throw new WrappedException(Layer.BUSINESS, e1);
318                    }
319                    spDescriptor.getKeyDescriptors().add(signingDescriptor);
320                    spDescriptor.getKeyDescriptors().add(encryptionDescriptor);
321                    
322                    spDescriptor.addSupportedProtocol(SAMLConstants.SAML20P_NS);
323                    spDescriptor.getAssertionConsumerServices().add(SAMLUtil.createAssertionConsumerService(baseUrl + "/SAMLAssertionConsumer", SAMLConstants.SAML2_POST_BINDING_URI, 0, true));
324                    if (enableArtifact) {
325                            spDescriptor.getAssertionConsumerServices().add(SAMLUtil.createAssertionConsumerService(baseUrl + "/SAMLAssertionConsumer", SAMLConstants.SAML2_ARTIFACT_BINDING_URI, 1, false));
326                    }
327                    if (enableRedirect) {
328                            spDescriptor.getAssertionConsumerServices().add(SAMLUtil.createAssertionConsumerService(baseUrl + "/SAMLAssertionConsumer", SAMLConstants.SAML2_REDIRECT_BINDING_URI, 2, false));
329                    }
330                    
331                    spDescriptor.getSingleLogoutServices().add(SAMLUtil.createSingleLogoutService(baseUrl + "/LogoutServiceHTTPRedirect", baseUrl + "/LogoutServiceHTTPRedirectResponse", SAMLConstants.SAML2_REDIRECT_BINDING_URI));
332                    
333                    if (enableSoap) {
334                            spDescriptor.getSingleLogoutServices().add(SAMLUtil.createSingleLogoutService(baseUrl + "/LogoutServiceSOAP", null, SAMLConstants.SAML2_SOAP11_BINDING_URI));
335                    }
336                    
337                    if(enablePostSLO) {
338                spDescriptor.getSingleLogoutServices().add(SAMLUtil.createSingleLogoutService(baseUrl + "/LogoutServiceHTTPPost", baseUrl + "/LogoutServiceHTTPRedirectResponse", SAMLConstants.SAML2_POST_BINDING_URI));
339                    }
340                    
341                    if (enableArtifact) {
342                            spDescriptor.getArtifactResolutionServices().add(SAMLUtil.createArtifactResolutionService(baseUrl + "/SAMLAssertionConsumer"));
343                    }
344                    
345                    if (supportOCESAttributes) {
346                            addAttributeConsumerService(spDescriptor, entityId);
347                    }
348                    
349                    descriptor.getRoleDescriptors().add(spDescriptor);
350                    return descriptor;
351            }

369                                    OIOSAMLConstants.ATTRIBUTE_CVR_NUMBER_IDENTIFIER_NAME,
370                                    OIOSAMLConstants.ATTRIBUTE_ORGANISATION_NAME_NAME,

374                                    OIOSAMLConstants.ATTRIBUTE_PSEUDONYM_NAME,
375                                    OIOSAMLConstants.ATTRIBUTE_USER_CERTIFICATE_NAME,

385                    }
386                    
387                    spDescriptor.getAttributeConsumingServices().add(service);
388            }

390            private List<?> extractParameterList(final HttpServletRequest request) {
391                    List<?> parameters;
392                    try {
393                            FileItemFactory itemFactory = new DiskFileItemFactory();
394                            parameters = new ServletFileUpload(itemFactory).parseRequest(request);
395                    } catch (FileUploadException e) {
396                            log.error("Unable to parse uploaded files", e);
397                            throw new RuntimeException("Unable to parse uploaded files", e);
398                    }
399                    return parameters;
400            }

402            private boolean checkConfiguration(HttpServletResponse response)
403                            throws IOException {
404                    if (isConfigured(servletContext)) {
405                            sendResponse(response, renderTemplate("alreadyConfigured.vm", new HashMap<String, Object>() {{
406                                    put("home", getHome(servletContext));
407                            }}, true));
408                            return false;
409                    }
410                    return true;
411            }

413            private FileItem extractFile(String name, List<?> files) {
414                    for (Iterator<?> i = files.iterator(); i.hasNext();) {
415                            FileItem file = (FileItem) i.next();
416                            if (!file.isFormField() && file.getFieldName().equals(name)) {
417                                    return file;
418                            }
419                    }
420                    return null;
421            }
422            
423            private String extractParameter(String name, List<?> files) {
424                    for (Iterator<?> i = files.iterator(); i.hasNext();) {
425                            FileItem file = (FileItem) i.next();
426                            if (file.isFormField() && file.getFieldName().equals(name)) {
427                                    return "".equals(file.getString()) ? null : file.getString();
428                            }
429                    }
430                    return null;
431            }

439            protected String getBaseUrl(HttpServletRequest request) {
440                    String url = request.getRequestURL().toString();
441                    int idx = url.lastIndexOf(request.getServletPath());
442                    
443                    return url.substring(0, idx + request.getServletPath().length());
444            }

446            protected boolean isHomeAvailable(ServletContext ctx) {
447                    String home = getHome(ctx);
448                    if (home == null) return false;
449                    
450                    if (new File(home).isDirectory()) {
451                            return true;
452                    } else {
453                            return false;
454                    }
455            }
456            
457            protected boolean isConfigured(ServletContext ctx) {
458                    String home = getHome(ctx);
459                    if (home == null) return false;
460                    
461                    File homeDir = new File(home);
462                    String[] files = homeDir.list();
463                    if (files != null && files.length > 0) {
464                            return true;
465                    } else {
466                            return false;
467                    }
468            }

470            protected String renderTemplate(String template, Map<String, Object> objects, boolean html) {
471                    VelocityContext ctx = new VelocityContext();
472                    for (Map.Entry<String, Object> e : objects.entrySet()) {
473                            ctx.put(e.getKey(), e.getValue());
474                    }
475                    
476                    StringWriter w = new StringWriter();
477                    
478                    try {
479                            String prefix = "/" + getClass().getPackage().getName().replace('.', '/') + "/";
480                            if (html) {
481                                    engine.mergeTemplate(prefix + "head.vm", ctx, w);
482                            }
483                            engine.mergeTemplate(prefix + template, ctx, w);
484                            if (html) {
485                                    engine.mergeTemplate(prefix + "foot.vm", ctx, w);
486                            }
487                    } catch (Exception e) {
488                            log.error("Unable to merge templates", e);
489                    }
490                    return w.toString();
491            }

493            private String getHome(ServletContext ctx) {
494                    String home = ctx.getInitParameter(Constants.INIT_OIOSAML_HOME);
495                    if (home == null) {
496                            home = System.getProperty(SAMLUtil.OIOSAML_HOME);
497                    }
498                    if (home == null) {
499                            String name = ctx.getInitParameter(Constants.INIT_OIOSAML_NAME);
500                            if (name != null) {
501                                    home = System.getProperty("user.home") + "/.oiosaml-" + name;
502                            }
503                    }
504                    if (home == null) {
505                            home = System.getProperty("user.home") + "/.oiosaml";
506                            File h = new File(home);
507                            if (h.exists() && !h.isDirectory()) {
508                                    throw new IllegalStateException(home + " is not a directory");
509                            } else if (!h.exists()) {
510                                    log.info("Creating empty config dir in " + home);
511                                    if (!h.mkdir()) {
512                                            throw new IllegalStateException(h + " could not be created");
513                                    }
514                            }
515                    }
516                    return home;
517            }

519            private String getEntityId(HttpServletRequest request) {
520                    return request.getScheme() + "://saml." + request.getServerName();
521            }

523            private boolean checkNotNull(Object ... objs) {
524                    for (Object o : objs) {
525                            if (o == null) {
526                                    return false;
527                            }
528                    }
529                    return true;
530            }

532            protected Map<String, Object> getStandardParameters(HttpServletRequest request) {
533                    String base = getBaseUrl(request);
534                    Map<String, Object> params = new HashMap<String, Object>();
535                    params.put("artifactResponseUrl", base + "/SAMLAssertionConsumer");
536                    params.put("postResponseUrl", base + "/SAMLAssertionConsumer");
537                    params.put("logoutUrl", base + "/SAMLAssertionConsumer");
538                    params.put("logoutResponseUrl", base + "/LogoutServiceHTTPRedirectResponse");
539                    params.put("logoutRequestUrl", base + "/LogoutServiceHTTPRedirect");
540                    params.put("logoutSoapRequestUrl", base + "/LogoutServiceSOAP");
541            params.put("logoutPostRequestUrl", base + "/LogoutServiceHTTPPost");
542                    params.put("home", getHome(servletContext));
543                    params.put("entityId", getEntityId(request));
544                    return params;
545            }
