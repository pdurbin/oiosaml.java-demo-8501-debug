024    package dk.itst.oiosaml.sp.model;
025    
026    import javax.xml.namespace.QName;
027    
028    import org.opensaml.xml.XMLObject;
029    import org.opensaml.xml.schema.XSAny;
030    import org.opensaml.xml.schema.impl.XSAnyBuilder;
031    
032    import dk.itst.oiosaml.common.OIOSAMLConstants;
033    
034    /**
035     * Class containing the assurance level associated with the login.
036     * The currently known assurance levels are:
037     * 2 - PASSWORD_ASSURANCE_LEVEL
038     * 3 - CERTIFICATE_ASSURANCE_LEVEL
039     * @author lsteinth
040     *
041     */
042    public class AssuranceLevel implements BRSSAMLExtensionObject {
043            public static final String VERSION = "$Id: AssuranceLevel.java 2829 2008-05-13 12:11:31Z jre $";
044        /** Element local name. */
045        public static final String DEFAULT_ELEMENT_LOCAL_NAME = "AssuranceLevel";
046    
047        /** Default element name. */
048        public static final QName DEFAULT_ELEMENT_NAME = new QName(OIOSAMLConstants.BRS_NS, DEFAULT_ELEMENT_LOCAL_NAME, OIOSAMLConstants.BRS_PREFIX);
049    
050        public static final int PASSWORD_ASSURANCE_LEVEL = 2;
051        public static final int CERTIFICATE_ASSURANCE_LEVEL = 3;
052        public static final int DEFAULT_ASSURANCE_LEVEL = PASSWORD_ASSURANCE_LEVEL;
053        
054            private int value;
055            /**
056             * Create a new assurance level with a given value
057             * @param value The value
058             */
059            public AssuranceLevel(String value) {
060                    try {
061                            this.value = Integer.parseInt(value);
062                    } catch (NumberFormatException e) {
063                            this.value = DEFAULT_ASSURANCE_LEVEL;
064                    }
065            }
066    
067            /**
068             * 
069             * @return The value associated with the assurance level
070             */
071            public int getValue() {
072                    return value;
073            }
074    
075            /**
076             * Set the value associated with the assurance level
077             * @param value The value
078             */
079            public void setValue(int value) {
080                    this.value = value;
081            }
082    
083            /**
084             * @see BRSSAMLExtensionObject#getXMLObject()
085             */
086            public XMLObject getXMLObject() {
087                    XSAnyBuilder builder = new XSAnyBuilder();
088                    XSAny ep = builder.buildObject(OIOSAMLConstants.BRS_NS, DEFAULT_ELEMENT_LOCAL_NAME, OIOSAMLConstants.BRS_PREFIX);
089                    ep.setTextContent(String.valueOf(value));
090                    return ep;
091            }
092    }
