024    package dk.itst.oiosaml.sp.model;
025    
026    import org.opensaml.saml2.core.StatusCode;
027    import org.opensaml.saml2.core.StatusResponseType;
028    
029    import dk.itst.oiosaml.sp.model.validation.ValidationException;
030    
031    public class OIOAbstractResponse extends OIOSamlObject {
032    
033            private final StatusResponseType response;
034    
035            public OIOAbstractResponse(StatusResponseType response) {
036                    super(response);
037                    this.response = response;
038            }
039    
040            /**
041             * @return true, if the issuer of the SAML Response match the expected
042             */
043            public boolean isIssuerOK(String issuer) {
044                    return response.getIssuer() != null
045                                    && response.getIssuer().getValue() != null
046                                    && response.getIssuer().getValue().equals(issuer);
047            }
048    
049            /**
050             * @param destination
051             *            The expected destination
052             * @return <code>true</code>, if the destination is match the destination in the
053             *         &lt;SAMLResponse&gt; - otherwise <code>false</code>. If the response does not have a destination, <code>true</code> is returned.
054             */
055            public boolean isDestinationOK(String destination) {
056                    if (response.getDestination() == null) return true;
057                    
058                    return response.getDestination() != null && response.getDestination().equals(destination);
059            }
060    
061            protected void validateResponse(String requestId, String expectedDestination, boolean allowPassive) {
062                    String statusCode = response.getStatus().getStatusCode().getValue();
063                    if (!StatusCode.SUCCESS_URI.equals(statusCode)) {
064                            
065                            StatusCode is = response.getStatus().getStatusCode().getStatusCode();
066                            if (is == null || !(StatusCode.NO_PASSIVE_URI.equals(is.getValue()) && allowPassive)) {
067                                    String msg = response.getStatus().getStatusMessage() == null ? "" : response.getStatus().getStatusMessage().getMessage();
068                                    throw new ValidationException("Got StatusCode " + statusCode + " should be " + StatusCode.SUCCESS_URI + ". Message: " + msg);
069                            }
070                    }
071                    if (!isDestinationOK(expectedDestination)) {
072                            throw new ValidationException("Wrong destination. Expected " + expectedDestination + ", was " + response.getDestination());
073                    }
074                    
075                    if (requestId != null && !requestId.equals(response.getInResponseTo())) {
076                            throw new ValidationException("Wrong InResponseTo. Expected " + requestId + ", was " + response.getInResponseTo());
077                    }
078    
079            }
080    
081            public String getInResponseTo() {
082                    return response.getInResponseTo();
083            }
084            
085            public String getID() {
086                    return response.getID();
087            }
088    
089                    
090    }
