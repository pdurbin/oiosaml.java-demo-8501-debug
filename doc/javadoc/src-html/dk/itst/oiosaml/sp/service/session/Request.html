024    package dk.itst.oiosaml.sp.service.session;
025    
026    import java.io.Serializable;
027    import java.util.Map;
028    
029    import javax.servlet.http.HttpServletRequest;
030    
031    public class Request implements Serializable {
032            
033            private final String requestURI;
034            private final String queryString;
035            private final String method;
036            private final Map<String, String[]> parameters;
037    
038            public Request(String requestURI, String queryString, String method, Map<String, String[]> parameters) {
039                    this.requestURI = requestURI;
040                    this.queryString = queryString;
041                    this.method = method;
042                    this.parameters = parameters;
043            }
044            
045            @SuppressWarnings("unchecked")
046            public static Request fromHttpRequest(HttpServletRequest req) {
047                    return new Request(req.getRequestURI(), req.getQueryString(), req.getMethod(), req.getParameterMap());
048            }
049            
050    
051            public String getMethod() {
052                    return method;
053            }
054            
055            public Map<String, String[]> getParameters() {
056                    return parameters;
057            }
058            
059            public String getQueryString() {
060                    return queryString;
061            }
062            
063            public String getRequestURI() {
064                    return requestURI;
065            }
066    }
