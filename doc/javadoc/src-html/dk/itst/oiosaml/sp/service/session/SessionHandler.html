024    package dk.itst.oiosaml.sp.service.session;
025    
026    import javax.servlet.http.HttpSession;
027    
028    import org.opensaml.saml2.core.Assertion;
029    
030    import dk.itst.oiosaml.sp.model.OIOAssertion;
031    
032    /**
033     * Handle session state across requests and instances.
034     * 
035     * <p>Due to SOAP Logout, it is not possible to store all state in the HTTP session. Instead, implementations of this interface handle session state,
036     * primarily based on the HTTP session.<p> 
037     * 
038     * <p>Implementations are expected to be thread-safe, and should not store any instance state, as a new instance will be created for
039     * every request.</p>
040     * 
041     * @see SessionHandlerFactory
042     */
043    public interface SessionHandler {
044    
045            /**
046             * Associate an assertion with a given session.
047             * 
048             * @throws IllegalArgumentException If the assertion is being replayed. Implementations must check that the assertion id has not been seen before.
049             */
050            public void setAssertion(String sessionId, OIOAssertion assertion) throws IllegalArgumentException;
051    
052            
053            /**
054             * @return true if the session is logged in and has a non expired assertion,
055             *         false otherwise. 
056             */
057            public boolean isLoggedIn(String sessionId);
058            
059            /**
060             * Mark a given session as it has been logged out by removing it the
061             * assertion from the session.
062             */
063            public void logOut(HttpSession session);
064            
065            /**
066             * Mark a given session as it has been logged out by removing it the associated
067             * assertion
068             * 
069             * @param sessionId
070             */
071            public void logOut(String sessionId);
072            
073            /**
074             * @return The {@link Assertion} associated with the session. <code>null</code> if there is no assertion.
075             */
076            public OIOAssertion getAssertion(String sessionId);
077            
078            /**
079             * @param sessionIndex
080             *            The sessionIndex from the assertion
081             * @return The sessionId associated with the sessionIndex in case there is
082             *         one, otherwise null
083             */
084            public String getRelatedSessionId(String sessionIndex);
085            
086            public void registerRequest(String id, String receiverEntityID);
087            
088            
089            /**
090             * Remove a request id from the list of registered request ids and return the registered IdP entity id.
091             * @param id
092             * @throws IllegalArgumentException If the request id is unknown.
093             */
094            public String removeEntityIdForRequest(String id) throws IllegalArgumentException;
095    
096            /**
097             * Clean stored ids and sessions.
098             * 
099             * @param requestIdsCleanupDelay Milliseconds to store assertion ids for replay prevention.
100             * @param sessionCleanupDelay Milliseconds to store session data before purging (in case logout has not been called).
101             */
102            public void cleanup(long requestIdsCleanupDelay, long sessionCleanupDelay);
103            
104            /**
105             * Set the max number of assertion ids to track for replay protection, and reset the cache.
106             * @param maxNum
107             */
108            public void resetReplayProtection(int maxNum);
109    
110            /**
111             * Save information about a request.
112             * 
113             * The information saved can be retrieved later on using getRequest to replay the request after the user has been authenticated.
114             * 
115             * @return A unique opaque string, no more than 72 characters long.
116             */
117            public String saveRequest(Request request);
118                    
119            /**
120             * Get the request for a state identifier.
121             * 
122             * @param state
123             * @throws IllegalArgumentException If the state identifier is unknown.
124             */
125            public Request getRequest(String state) throws IllegalArgumentException;
126    }
