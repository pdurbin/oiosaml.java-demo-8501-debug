024    package dk.itst.oiosaml.sp.service.util;
025    
026    import java.io.IOException;
027    
028    import org.opensaml.ws.soap.soap11.Envelope;
029    import org.opensaml.xml.XMLObject;
030    
031    import dk.itst.oiosaml.common.SOAPException;
032    import dk.itst.oiosaml.sp.model.OIOSamlObject;
033    
034    public interface SOAPClient {
035            public Envelope wsCall(XMLObject obj, String location, String username, String password, boolean ignoreCertPath) throws IOException;
036            public XMLObject wsCall(OIOSamlObject obj, String location, String username, String password, boolean ignoreCertPath) throws IOException;
037    
038            /**
039             * Execute a raw request against a SOAP service.
040             * @param location HTTP Endpoint for the service.
041             * @param username Username for http basic auth. Can be <code>null</code>, in which case basic auth is disabled.
042             * @param password Basic auth password.
043             * @param ignoreCertPath Set to <code>true</code> to ignore certificate path errors on ssl connections.
044             * @param xml Complete SOAP Envelope as string.
045             * @param soapAction SOAP Action to invoke.
046             * @return The response envelope if the server returned 200 OK. An exeption is thrown in all other cases.
047             * @throws IOException If a generic IO exception occurred, for example if the connection failed.
048             * @throws SOAPException If the server returned anything but 200 OK.
049             */
050            public Envelope wsCall(String location, String username, String password, boolean ignoreCertPath, String xml, String soapAction) throws IOException, SOAPException;
051    }
