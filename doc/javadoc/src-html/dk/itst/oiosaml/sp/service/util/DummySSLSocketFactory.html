001    package dk.itst.oiosaml.sp.service.util;
002    
003    import java.io.IOException;
004    import java.net.InetAddress;
005    import java.net.Socket;
006    
007    import javax.net.SocketFactory;
008    import javax.net.ssl.SSLContext;
009    import javax.net.ssl.SSLSocketFactory;
010    import javax.net.ssl.TrustManager;
011    
012    public class DummySSLSocketFactory extends SSLSocketFactory {
013            private SSLSocketFactory factory;
014    
015            public DummySSLSocketFactory() {
016                    try {
017                            SSLContext sslcontext = SSLContext.getInstance("TLS");
018                            sslcontext.init(null,
019                                            new TrustManager[] { new DummyTrustManager()},
020                                            null);
021                            factory = (SSLSocketFactory)sslcontext.getSocketFactory();
022                    } catch(Exception ex) {
023                            // ignore
024                    }
025            }
026    
027            public static SocketFactory getDefault() {
028                    return new DummySSLSocketFactory();
029            }
030    
031            public Socket createSocket(Socket socket, String s, int i, boolean flag)
032            throws IOException {
033                    return factory.createSocket(socket, s, i, flag);
034            }
035    
036            public Socket createSocket(InetAddress inaddr, int i,
037                            InetAddress inaddr1, int j) throws IOException {
038                    return factory.createSocket(inaddr, i, inaddr1, j);
039            }
040    
041            public Socket createSocket(InetAddress inaddr, int i)
042            throws IOException {
043                    return factory.createSocket(inaddr, i);
044            }
045    
046            public Socket createSocket(String s, int i, InetAddress inaddr, int j)
047            throws IOException {
048                    return factory.createSocket(s, i, inaddr, j);
049            }
050    
051            public Socket createSocket(String s, int i) throws IOException {
052                    return factory.createSocket(s, i);
053            }
054    
055            public String[] getDefaultCipherSuites() {
056                    return factory.getDefaultCipherSuites();
057            }
058            public Socket createSocket() throws IOException {
059                    return factory.createSocket();
060            }
061            public String[] getSupportedCipherSuites() {
062                    return factory.getSupportedCipherSuites();
063            }
064    }
