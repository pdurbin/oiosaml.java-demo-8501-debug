024    package dk.itst.oiosaml.sp.service;
025    
026    import java.io.IOException;
027    
028    import javax.servlet.ServletException;
029    
030    public class MetadataHandler implements SAMLHandler {
031    
032            public void handleGet(RequestContext context) throws ServletException, IOException {
033                    if (context.getRequest().getParameter("raw") != null) {
034                            context.getResponse().setContentType("text/plain");
035                    } else {
036                            context.getResponse().setCharacterEncoding("utf-8");
037                    }
038                    
039                    boolean sign = context.getRequest().getParameter("unsigned") == null;
040                    String metadata = context.getSpMetadata().getMetadata(context.getCredential(), sign);
041                    
042                    context.getResponse().getWriter().print(metadata);
043            }
044    
045            public void handlePost(RequestContext context) throws ServletException, IOException {
046                    throw new UnsupportedOperationException("POST not allowed");
047            }
048    
049    }
