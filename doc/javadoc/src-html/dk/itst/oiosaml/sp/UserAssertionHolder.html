024    package dk.itst.oiosaml.sp;
025    
026    /**
027     * ThreadLocal holder for the current user assertion.
028     * 
029     * Use {@link #get()} to retrieve the assertion from the application layer if there is no access to the session.
030     *  
031     * @author recht
032     *
033     */
034    public  final class UserAssertionHolder {
035            private static final ThreadLocal<UserAssertion> assertion = new ThreadLocal<UserAssertion>();
036    
037            /**
038             * Get the current user assertion. Will return <code>null</code> if the user is not authenticated.
039             */
040            public static UserAssertion get() {
041                    return assertion.get();
042            }
043            
044            public static void set(UserAssertion assertion) {
045                    UserAssertionHolder.assertion.set(assertion);
046            }
047    }
