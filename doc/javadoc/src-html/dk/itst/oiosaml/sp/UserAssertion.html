024    package dk.itst.oiosaml.sp;
025    
026    import java.security.cert.X509Certificate;
027    import java.util.Collection;
028    import java.util.Date;
029    
030    public interface UserAssertion {
031            
032            /**
033             * @return Whether the Assertion is compliant with OIOSAML version 2.0.5.
034             */
035            public boolean isOIOSAMLCompliant();
036            
037            /**
038             * @return Whether the Assertion is in compliance with the OCES profile in OIOSAML version 2.0.5
039             */
040            public boolean isOCESProfileCompliant();
041            
042            /**
043             * Persistent pseudonyms are used to support federation using persistent
044             * pseudonym identifiers. A pseudonym identifier is in effect a random value that an IdP-
045             * SP pair establishes and uses to refer to the same user. The shared identifier must be
046             * unique to the actual IdP-SP pairing. Each entity maintains a mapping from the shared
047             * identifier to their internal representation. The goal of this attribute profile is to define
048             * the content of assertions and attributes supporting this scenario.
049             * @return Whether the Assertion is an persistent pseudonym.
050             */
051            public boolean isPersistentPseudonymProfileCompliant();
052            
053            /**
054             * @return Raw xml representation of the SAML assertion.
055             **/
056            public String getXML();
057            
058            /**
059             * @return Expiretime for the users session with the Identity Provider (IdP). May be null if IdP does not provide it.
060             */
061            public Date getSessionExpireTime();
062    
063            /**
064             * @return The time of issue of the assertion.
065             */     
066            public Date getIssueTime();
067    
068            /**
069             * @return The ID of the Identity Provider
070             */
071            public String getIssuer();
072            
073            /**
074             * @return Whether the assertion is signed. If signed, the signature is also guaranteed to be valid.
075             */
076            public boolean isSigned();
077            
078            /**
079             * @return The Name ID value of the subject Node. Often just the username. Is always present. 
080             */
081            public String getSubject();
082            
083            /**
084             * @return The format of the subject name id. Could be: urn:oasis:names:tc:SAML:1.1:nameid-format:unspecified <br/>
085             * @see #getSubject()
086             */
087            public NameIDFormat getNameIDFormat();
088    
089            /**
090             * The surname of the subject, attribute: urn:oid:2.5.4.4 <br/>
091             * Mandatory in DK-SAML 2.0.5
092             * @return The Surname of the subject.
093             */
094            public String getSurname();
095            
096            /**
097             * The common name of the subject, attribute: urn:oid:2.5.4.3 <br/>
098             * Mandatory in DK-SAML 2.0.5
099             *       * @return The common name of the subject. 
100             */
101            public String getCommonName();
102            
103            /**
104             * The user id attribute of the subject, attribute: urn:oid:0.9.2342.19200300.100.1.1<br/>
105             * The uid attribute specifies the user id in the user s (principal s) home organization (or
106             * credential issuing organization where home organization is unknown or doesn't exist
107             * which is the case for citizens).<br />
108             * The actual content of the uid attribute is left to the discretion of the IdP, and should be
109             * documented by the IdP, unless an OCES certificate was used by the user at login, in which
110             * case the following applies:<br />
111             * In the OCES attribute profile, the following conventions apply for the uid attribute:
112             * The uid attribute must contain the Subject Serial number from the OCES
113             * certificate. The field from the certificate is included literally.
114             * This means that the PID and RID numbers will be present twice in the assertion, but
115             * this may be convenient:<br />
116             * If the Service Provider needs a unique ID within the credential issuing
117             * organization or he needs the Subject Serial Number, he may simply pick the
118             * uid attribute.<br />
119             * If the Service Provider wants to know whether the Subject is a person or
120             * employee or needs the RID/PID/CPR/CVR numbers, he can pick the
121             * corresponding (atomic) attributes without having to parse the serial number
122             * string.<br />
123             *       Mandatory in DK-SAML 2.0.5 
124             * @return The user id of the subject. 
125             */
126            public String getUserId();
127            
128            /**
129             * The email of the subject, attribute: urn:oid:0.9.2342.19200300.100.1.3 <br/>
130             *       Mandatory in DK-SAML 2.0.5 
131             * @return The email of the subject. 
132             */
133            public String getMail();
134    
135            /**
136             * The Assurance level of the subject, attribute:  dk:gov:saml:attribute:AssuranceLevel<br/>
137             *  In the xml representation the value may be the string "1", "2", "3", "4", "test", represented here as integers, respectively 1, 2, 3, 4, -1<br />
138             *       Mandatory in DK-SAML 2.0.5 <br/>
139             * <a href="http://www.itst.dk/arkitektur-og-standarder/Standardisering/standarder-for-serviceorienteret-infrastruktur/standarder-for-brugerstyring/resolveuid/c180053b937f49169aa82b6d4a839620">Standard defining semantics of assurancelevel</a>
140             *  <ul><li> Niveau 1 - Lille eller ingen tiltro til påstået identitet</li>
141             *   <li>Niveau 2 - Nogen tiltro til påstået identitet</li>
142             *   <li>Niveau 3 - Høj tillid til påstået identitet</li>
143             *   <li>Niveau 4 - Meget høj tillid til påstået identitet</li></ul>
144    
145             * @return The assurance level of the SAML assertion. Defined in "Vejledning vedrørende niveauer af autenticitetssikring"
146             */
147            public int getAssuranceLevel();
148    
149            /**
150             * The specification version, attribute: dk:gov:saml:attribute:SpecVer.<br />
151             * The SpecVer attribute tells the Service Provider which version of the DK-SAML
152             * profile the assertion was issued under.
153             * @return The specification version of the assertion. 
154             */
155            public String getSpecificationVersion();
156    
157            /**
158             * The unique account key of the subject, attribute: dk:gov:saml:attribute:UniqueAccountKey.<br />
159             * <b>Not</b> Mandatory in DK-SAML 2.0.5<br />
160             * Subject to these <a href="http://www.itst.dk/arkitektur-og-standarder/Standardisering/standarder-for-serviceorienteret-infrastruktur/standarder-for-brugerstyring/resolveuid/a3d519d0ce1f304eb1b4fc6f7941b027">recommendations</a> 
161             * @return The unique account key of the subject if present, null otherwise 
162             */
163            public String getUniqueAccountKey();
164            
165            /**
166             * The CVR identifier of the subject, attribute: dk:gov:saml:attribute:CvrNumberIdentifier<br />
167             * <b>Not</b> Mandatory in DK-SAML 2.0.5 
168             * @return The CPR identifier of the subject if present, null otherwise. 
169             */
170            public String getCVRNumberIdentifier();
171                            
172            /**
173             * Get an attribute on the assertion by the name of the assertion.<br />
174             * @param name of the assertion, e.g. "dk:gov:saml:attribute:CvrNumberIdentifier"
175             * @return the requested attribute (immutable), null if not present.
176             */
177            public UserAttribute getAttribute(String name);
178            
179            /**
180             * Get a list of all attributes in the assertion.
181             * @return Immutable list containing all attributes.
182             */
183            public Collection<UserAttribute> getAllAttributes();
184            
185            /**
186             * Attribute name: urn:oid:2.5.4.5<br />
187             * Mandatory in the OCES profile of DK-SAML 2.0.5
188             * @return The certificate serial number
189             */
190            public String getCertificateSerialNumber();
191            
192            /**
193             * Attribute name: urn:oid:2.5.4.10<br />
194             * Mandatory for companies and employees in the OCES profile of DK-SAML 2.0.5
195             * @return The organization name
196             */
197            public String getOrganizationName(); 
198            
199            /**
200             * The name of the department within an organisation. Attribute name: urn:oid:2.5.4.11<br />
201             * <b>Not</b> mandatory in the OCES profile of DK-SAML 2.0.5
202             * @return The organisation unit.
203             */     
204            public String getOrganizationUnit();
205            
206            /**
207             * Attribute name: urn:oid:2.5.4.12<br />
208             * <b>Not</b> mandatory in the OCES profile of DK-SAML 2.0.5
209             * @return the title of the subject (as employee in an organisation)
210             **/
211            public String getTitle();
212            
213            /**
214             * Attribute name: urn:oid:2.5.4.16<br />
215             * <b>Not</b> mandatory in the OCES profile of DK-SAML 2.0.5
216             * @return the postal address of a company or person.
217             */
218            public String getPostalAddress();
219    
220            /**
221             * Attribute name: urn:oid:2.5.4.65<br />
222             * <b>Not</b> mandatory in the OCES profile of DK-SAML 2.0.5
223             * @return pesudonym
224             */     
225            public String getPseudonym();
226            
227            /**
228             * Attribute name: dk:gov:saml:attribute:IsYouthCert<br />
229             * Mandatory in the OCES profile of DK-SAML 2.0.5
230             * @return TRUE if the certificate is a youth certificate, FALSE if not, and null if the attribute was not present.
231             */
232            public Boolean isYouthCertificate();
233            
234            /**
235             * Attribute name: urn:oid:1.3.6.1.4.1.1466.115.121.1.8<br />
236             * <b>Not</b> mandatory in the OCES profile of DK-SAML 2.0.5
237             * @return The OCES user certificate if provided by the IdP, null otherwise.
238             */
239            public X509Certificate getUserCertificate();
240            
241            /**
242             * For OCES person certificates, the most interesting attribute is the PID number which
243             * contains a unique identifier for the person6. The advantage of PID numbers over CPR
244             * numbers is that they can be freely exchanged without risk of violating personal data
245             * protection acts.<br/>
246             * Attribute name: dk:gov:saml:attribute:PidNumberIdentifier<br />
247             * Mandatory in the OCES profile of DK-SAML 2.0.5
248             * @return The PID number from the OCES certificate presented to the IdP  
249             */
250            public String getPIDNumber();
251            
252            /**
253             * Attribute name: dk:gov:saml:attribute:CprNumberIdentifier<br />
254             * In some scenarios, it may be easier to transfer the CPR number directly in the
255             * assertion. The CPR number attribute is optional and must only be included when:<br/>
256         * <ul><li> A formal agreement has been made to exchange it</li>
257         *   <li>The Service Provider is authorized to receive it (e.g. is a Government entity)</li>
258         *   <li>The surrounding assertion is encrypted (which is mandatory in this profile)</li></ul>
259             * An Identity Provider must have the technical capability to resolve and insert the CPR
260             * number both for citizens and employees who have one7. The CPR number attribute is
261             * however optional such that it can be omitted from assertions for Service Providers
262             * who do not need it / are not allowed receiving it.<br />
263             * <b>Not</b> mandatory in the OCES profile of DK-SAML 2.0.5
264             * @return The CPR number of the subject if available, null otherwise.
265             */
266            public String getCPRNumber();
267            
268            /**
269             * Attribute name: dk:gov:saml:attribute:RidNumberIdentifier<br /> 
270             * This attribute is mandatory when the user has authenticated with an employee
271             * certificate (syntax and semantics of the number is defined in DS844).<br />
272             * Mandatory for employees in the OCES profile of DK-SAML 2.0.5 
273             * @return The employee number / PID of the employee if present, null otherwise.
274             */
275            public String getRIDNumber();
276            
277            /**
278             * Returns the ID of the assertion.
279             */
280            public String getAssertionId();
281            
282            /**
283             * Checks if the current user is authenticated.
284             * @return <code>true</code> if a valid assertion has been received. <code>false</code> if a valid assertion has not been received, but
285             * the user has been allowed to continue. This can happen if IsPassive is set, and the user was signed on.
286             */
287            public boolean isAuthenticated();
288            
289    }
