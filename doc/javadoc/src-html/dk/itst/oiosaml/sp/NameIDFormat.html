024    package dk.itst.oiosaml.sp;
025    
026    /**
027     * Standard SAML 2.0 Name ID formats.
028     * 
029     * @author recht
030     *
031     */
032    public enum NameIDFormat {
033            UNSPECIFIED("urn:oasis:names:tc:SAML:1.1:nameid-format:unspecified"),
034            EMAIL("urn:oasis:names:tc:SAML:1.1:nameid-format:emailAddress"),
035            X509SUBJECT("urn:oasis:names:tc:SAML:1.1:nameid-format:X509SubjectName"),
036            WINDOWS_DOMAIN("urn:oasis:names:tc:SAML:1.1:nameid-format:WindowsDomainQualifiedName"),
037            KERBEROS_PRINCIPAL("urn:oasis:names:tc:SAML:2.0:nameid-format:kerberos"),
038            ENTITY("urn:oasis:names:tc:SAML:2.0:nameid-format:entity"),
039            PERSISTENT("urn:oasis:names:tc:SAML:2.0:nameid-format:persistent"),
040            TRANSIENT("urn:oasis:names:tc:SAML:2.0:nameid-format:transient");       
041    
042            private final String format;
043            
044            private NameIDFormat(String format) {
045                    this.format = format;
046            }
047            
048            @Override
049            public String toString() {
050                    return super.toString() + ": " + format;
051            }
052            
053            /**
054             * Get the enumeration value from a format URI.
055             * 
056             * @param format SAML 2.0 format.
057             * @throws IllegalArgumentException If the format is not defined in SAML 2.0.
058             */
059            public static NameIDFormat getNameID(String format) {
060                    for (NameIDFormat id : values()) {
061                            if (id.format.equals(format)) {
062                                    return id;
063                            }
064                    }
065                    throw new IllegalArgumentException("Format " + format + " unknown");
066            }
067    
068            public String getFormat() {
069                    return format;
070            }
071    }
