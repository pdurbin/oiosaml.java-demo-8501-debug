108            /**
109             * Build a new empty object of the requested type.
110             * 
111             * The requested type must have a DEFAULT_ELEMENT_NAME attribute describing the element type as a QName.
112             * 
113             * @param <T> SAML Object type
114             */
115            @SuppressWarnings("unchecked")
116            public static <T extends XMLObject> T buildXMLObject(Class<T> type) {
117                    try {
118                            QName objectQName = getElementQName(type);
119                            XMLObjectBuilder<T> builder = Configuration.getBuilderFactory().getBuilder(objectQName);
120                            if (builder == null) {
121                                    throw new InvalidParameterException("No builder exists for object: " + objectQName.getLocalPart());
122                            }
123                            return builder.buildObject(objectQName.getNamespaceURI(), objectQName.getLocalPart(), objectQName.getPrefix());
124                    } catch (SecurityException e) {
125                            throw new RuntimeException(e);
126                    }                               
127            }
128    
129            private static <T> QName getElementQName(Class<T> type) {
130                    if (elementCache.containsKey(type)) return elementCache.get(type);
131                    
132                    try {
133                            Field typeField;
134                            try { 
135                                    typeField = type.getDeclaredField("DEFAULT_ELEMENT_NAME");
136                            } catch (NoSuchFieldException ex) {
137                                    typeField = type.getDeclaredField("ELEMENT_NAME");
138                            }
139    
140                            QName objectQName = (QName) typeField.get(null);
141                            elementCache.put(type, objectQName);
142                            return objectQName;
143                    } catch (NoSuchFieldException e) {
144                            throw new RuntimeException(e);
145                    } catch (IllegalAccessException e) {
146                            throw new RuntimeException(e);
147                    }
148            }

193             * Create a subject with a given nameID value.
194             * 
195             * The subject is given a confirmation with method bearer.
196             * 
197             * @param nameIDValue
198             *            The value of the nameID
199             * @return The SAML subject with the given nameId
200             */
201            public static Subject createSubject(String nameIDValue,
202                            String recipient, DateTime notOnOrAfter) {
203                    Subject subject = buildXMLObject(Subject.class);
204                    subject.setNameID(createNameID(nameIDValue));
205                    SubjectConfirmation subjectConfirmation = buildXMLObject(SubjectConfirmation.class);
206                    subjectConfirmation.setMethod(OIOSAMLConstants.METHOD_BEARER);
207                    SubjectConfirmationData subjectConfirmationData = buildXMLObject(SubjectConfirmationData.class);
208                    subjectConfirmationData.setRecipient(recipient);
209                    subjectConfirmationData.setNotOnOrAfter(notOnOrAfter);
210                    subjectConfirmation.setSubjectConfirmationData(subjectConfirmationData);
211                    subject.getSubjectConfirmations().add(subjectConfirmation);
212                    return subject;
213            }

322             * Unmarshall a resource file containing a SAML2.0 document in XML to an XMLObject.
323             * 
324             * @return The corresponding {@link XMLObject}
325             */
326            public static XMLObject unmarshallElement(InputStream input) {
327                    try {
328                            Element samlElement = loadElement(input);
329    
330                            Unmarshaller unmarshaller = Configuration.getUnmarshallerFactory().getUnmarshaller(samlElement);
331                            if (unmarshaller == null) {
332                                    log.error("Unable to retrieve unmarshaller by DOM Element");
333                                    throw new IllegalArgumentException("No unmarshaller for " + samlElement);
334                            }
335    
336                            return unmarshaller.unmarshall(samlElement);
337                    } catch (UnmarshallingException e) {
338                            log.error("Unmarshalling failed when parsing element file " + input, e);
339                    }
340    
341                    return null;
342            }

345                    Unmarshaller unmarshaller = Configuration.getUnmarshallerFactory().getUnmarshaller(element);
346                    if (unmarshaller == null) {
347                            log.error("Unable to retrieve unmarshaller by DOM Element " + element);
348                            throw new IllegalArgumentException("No unmarshaller for " + element);

351                            return unmarshaller.unmarshall(element);
352                    } catch (UnmarshallingException e) {
353                            log.error("Unmarshalling failed when parsing element file " + element, e);
354                            return null;

358            /**
359             * Read the content of a given XML resource file.
360             * 
361             * @param input
362             *            The stream to read from. The stream is not closed.
363             * @return The corresponding {@link Element}
364             */
365            public static Element loadElement(InputStream input) {
366                    try {
367                DocumentBuilderFactory newFactory = DocumentBuilderFactory.newInstance();
368                newFactory.setNamespaceAware(true);
369                
370                DocumentBuilder builder = newFactory.newDocumentBuilder();
371    
372                            Document doc = builder.parse(input);
373                            Element samlElement = doc.getDocumentElement();
374    
375                            return samlElement;
376                    } catch (ParserConfigurationException e) {
377                            log.error("Unable to parse element file " + input, e);
378                    } catch (SAXException e) {
379                            log.error("Unable to parse element file " + input, e);
380                    } catch (IOException e) {
381                            log.error("Unable to parse element file " + input, e);
382                    }
383                    return null;
384            }

411             * Parse an XML string.
412             * 
413             * @param elementString
414             *            The String to parse
415             * @return The corresponding document {@link Element}.
416             */
417            public static Element loadElementFromString(String elementString) {
418                    try {
419                DocumentBuilderFactory newFactory = DocumentBuilderFactory.newInstance();
420                newFactory.setNamespaceAware(true);
421                
422                DocumentBuilder builder = newFactory.newDocumentBuilder();
423    
424                            Document doc = builder.parse(new ByteArrayInputStream(elementString.getBytes("UTF-8")));

428                    } catch (ParserConfigurationException e) {

430                            throw new WrappedException(Layer.DATAACCESS, e);
431                    } catch (SAXException e) {
432                            log.error("Ue, nable to parse element string " + elementString, e);
433                            throw new WrappedException(Layer.DATAACCESS, e);
434                    } catch (IOException e) {
435                            log.error("Unable to parse element string " + elementString, e);
436                            throw new WrappedException(Layer.DATAACCESS, e);
437                    }
438            }

440            /**
441             * Unmarshall the content of a file containing a SAML2.0 document in XML to
442             * an XMLObject.
443             * 
444             * @param fileName
445             *            The name of the file from the file system.
446             * @return The corresponding {@link XMLObject}. Returns <code>null</code> if the file does not exist or is malformed.
447             */
448            public static XMLObject unmarshallElementFromFile(String fileName) {
449                    File file = new File(fileName);
450                    if (!file.isFile() || !file.canRead()) {
451                            log.error("Can't find or read file " + fileName);
452                            throw new RuntimeException("Cannot find file " + fileName);
453                    }
454    
455                    try {
456                            Element samlElement = loadElementFromFile(fileName);
457    
458                            Unmarshaller unmarshaller = Configuration.getUnmarshallerFactory()
459                                            .getUnmarshaller(samlElement);
460                            if (unmarshaller == null) {
461                                    log.error("Unable to retrieve unmarshaller by DOM Element for {" + samlElement.getNamespaceURI() + "}" + samlElement.getLocalName());
462                                    throw new IllegalArgumentException("No unmarshaller for element {" + samlElement.getNamespaceURI() + "}" + samlElement.getLocalName() + " from file " + fileName);
463                            }
464    
465                            return unmarshaller.unmarshall(samlElement);
466                    } catch (UnmarshallingException e) {
467                            throw new WrappedException(Layer.DATAACCESS, e);
468                    }
469            }

471            /**
472             * Read the content of a given XML file.
473             * 
474             * @param fileName
475             *            The name of the file
476             * 
477             * @return The corresponding {@link Element}
478             */
479            public static Element loadElementFromFile(String fileName) {
480                    try {
481                            StringBuffer sb = new StringBuffer(2048);
482                            BufferedReader reader = new BufferedReader(new FileReader(fileName));
483    
484                            char[] chars = new char[1024];
485                            int numRead = 0;
486                            while ((numRead = reader.read(chars)) != -1) {
487                                    String readData = String.valueOf(chars, 0, numRead);
488                                    sb.append(readData);
489                                    chars = new char[1024];
490                            }
491                            reader.close();
492                            if (log.isDebugEnabled()) log.debug(sb.toString());
493                            return loadElementFromString(sb.toString());
494                    } catch (IOException e) {
495                            throw new WrappedException(Layer.DATAACCESS, e);
496                    }
497            }

515            public static Element marshallObject(XMLObject object) {
516                    if (object.getDOM() == null) {
517                            Marshaller m = (Marshaller) Configuration.getMarshallerFactory().getMarshaller(object);
518                            if (m == null) {
519                                    throw new IllegalArgumentException("No unmarshaller for " + object);
520                            }
521                            try {
522                                    return m.marshall(object);
523                            } catch (MarshallingException e) {
524                                    throw new WrappedException(Layer.CLIENT, e);
525                            }
526                    } else {
527                            return object.getDOM();
528                    }
529            }

582            /**
583             * Decode the value of a _saml_idp discovery value.
584             * 
585             * @param value A string of entityIds encoded with base64 and separated by space.
586             * @return A list of decoded values. Never <code>null</code>.
587             */
588            public static String[] decodeDiscoveryValue(String value) {
589                    if (value == null) {
590                            return new String[0];
591                    }
592                    String[] ids = value.split(" ");
593                    for (int i = 0; i < ids.length; i++) {
594                            ids[i] = new String(Base64.decode(ids[i]));
595                    }
596                    return ids;
597            }

605            /**
606             * Get the first element of a specific type from a parent element.
607             * @param obj The parent element. If this is <code>null</code>, <code>null</code> is returned.
608             * @param type The type to retrieve.
609             * @return The first element, or <code>null</code> if no elements were found.
610             */
611            @SuppressWarnings("unchecked")
612            public static <T extends XMLObject> T  getFirstElement(ElementExtensibleXMLObject obj, Class<T> type) {
613                    if (obj == null) return null;
614                    
615                    for (XMLObject o : obj.getUnknownXMLObjects()) {
616                            if (type.isInstance(o)) {
617                                    return (T) o;
618                            }
619                    }
620                    return null;
621            }
